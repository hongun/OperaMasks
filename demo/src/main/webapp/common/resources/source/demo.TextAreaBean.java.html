package demo;

import java.io.Serializable;

import org.operamasks.faces.annotation.ManagedBean;
import org.operamasks.faces.annotation.ManagedBeanScope;

@ManagedBean(scope=ManagedBeanScope.SESSION)
public class TextAreaBean implements Serializable {
    private int width = 200;
    private int growMin = 60;
    private int growMax = 100;
    private boolean grow = true;
    private boolean preventScrollbars = true;
    public int getWidth() {
        return width;
    }
    public void setWidth(int width) {
        this.width = width;
    }
    public int getGrowMin() {
        return growMin;
    }
    public void setGrowMin(int growMin) {
        this.growMin = growMin;
    }
    public int getGrowMax() {
        return growMax;
    }
    public void setGrowMax(int growMax) {
        this.growMax = growMax;
    }
    public boolean isGrow() {
        return grow;
    }
    public void setGrow(boolean grow) {
        this.grow = grow;
    }
    public boolean isPreventScrollbars() {
        return preventScrollbars;
    }
    public void setPreventScrollbars(boolean preventScrollbars) {
        this.preventScrollbars = preventScrollbars;
    }
    
}
